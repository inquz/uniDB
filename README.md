create table if not exists public.workshops (
  workshopid integer generated by default as identity primary key,
  workshopname text not null unique
);

create table if not exists public.products (
  productid integer generated by default as identity primary key,
  productcode text not null unique,
  productname text not null,
  costprice numeric(12,2) not null check (costprice >= 0),
  wholesaleprice numeric(12,2) not null check (wholesaleprice >= 0),
  check (productcode ~ '^[A-Z]{2}[0-9]{5}$')
);

create table if not exists public.releases (
  workshopid integer not null references public.workshops(workshopid) on delete cascade,
  productid integer not null references public.products(productid) on delete cascade,
  quantity integer not null check (quantity > 0),
  primary key (workshopid, productid)
);

create or replace view public.workshop_report as
select
  w.workshopid,
  w.workshopname,
  coalesce(sum(r.quantity * p.costprice), 0)::numeric(14,2) as total_cost,
  coalesce(sum(r.quantity * p.wholesaleprice), 0)::numeric(14,2) as total_revenue,
  case
    when coalesce(sum(r.quantity * p.costprice), 0) = 0 then 0
    else round(
      coalesce(sum(r.quantity * p.wholesaleprice), 0)
      / nullif(sum(r.quantity * p.costprice), 0),
      2
    )
  end as efficiency_ratio
from public.workshops w
left join public.releases r on r.workshopid = w.workshopid
left join public.products p on p.productid = r.productid
group by w.workshopid, w.workshopname;

create or replace view public.top_products as
select
  p.productid,
  p.productcode,
  p.productname,
  sum(r.quantity) as total_quantity
from public.products p
join public.releases r on r.productid = p.productid
group by p.productid, p.productcode, p.productname
order by total_quantity desc, p.productid
limit 3;
